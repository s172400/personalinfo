function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    if (sheet.getLastRow() === 0) {
      sheet.appendRow(['제출일시','서비스명','성명','연락처','생년월일',
                       '수집이용','제3자제공','고유식별','마케팅']);
    }
    var p = e.parameter;
    sheet.appendRow([p.submittedAt||'', p.serviceName||'', p.name||'',
      p.phone||'', p.dob||'', p.collection||'',
      p.thirdParty||'', p.uniqueId||'', p.marketing||'']);
  } catch(err) { Logger.log(err.toString()); }
  return HtmlService.createHtmlOutput('');
}
function doGet(e) { return HtmlService.createHtmlOutput('OK'); }
